			SELECT CueVari.id, convert(int,CueVari.idcuepedido) as idcuepedido
			,convert(int,CueVari.codigoArticulo) as articulo_numero
			,convert(int,CueVari.codigoVariedad) as variedad_codigo
			,CueVari.cantidad
			FROM CabPedido as CabePedido
			Left Join CuePedido on CabePedido.id = CuePedido.idcabpedido
			Inner Join CuePedidoVariedad as CueVari on CuePedido.id = CueVari.idcuepedido
			Where CabePedido.id in (126976)
			order by cuevari.id
