	Select 0 AS ESTADO, CsrMovCtacte.*,Csrmaopera.idcomproba as idcomproba,Csrmaopera.numcomp as numcomp,CsrMaopera.clasecomp
	,(select ISNULL(SUM(CsrTablaimp.importe),0) from Tablaimp as CsrTablaimp 
		Where Csrmovctacte.idmaopera=Csrtablaimp.idmaopera and Csrtablaimp.tipoconce in ('NG')) Gravado
	,ISNULL(CsrCabeCpra.idprovrecep,isnull(CsrCabeCpra.idprovincia,ISNULL(CsrCabefac.idprovincia,cast(0 as int)))) as idprovincia
	,(CsrComprobante.regish+CsrComprobante.regisa) as compacreedor
	,ISNULL(CsrCentrorecep.abrevia,SPACE(3)) as CentroRecep
	,ISNULL(CsrProvincia.nombre,SPACE(15)) as ProvRecep
	from Movctacte as CsrMovCtacte
	left join maopera as Csrmaopera on CsrMovctacte.idmaopera = Csrmaopera.id
	left join cabecpra as CsrCabecpra on CsrMaopera.id = CsrCabecpra.idmaopera
	LEFT JOIN cabefac as CsrCabefac on CsrMaopera.id = CsrCabefac.idmaopera
	left join comprobante as Csrcomprobante on CsrMaopera.idcomproba = CsrComprobante.id
	left join CentroRecep as CsrCentroRecep on CsrCabeCpra.idcentrorecep = CsrCentroRecep.id
	left join Provincia as CsrProvincia on CsrCentroRecep.idprovincia = CsrProvincia.id
	where CsrmovCtacte.idctacte=  1100000004 and CsrMovctacte.saldo<>0 
	and Csrmaopera.idcomproba in (1,2,7,8,17,18,19,22,27,29,31,32,35,36,46,47,44,1100000049,1100000060,62,3,4,7,8,16,21,22,28,29,33,34,37,38,46,47,44,1100000049,62) 
	and Csrmaopera.origen in ('COB','FAC','FPE','RFL','IMP','PAG','CPR','IMP','CHZ') 
	and Csrmaopera.numcomp<> ' ' and LEFT(CsrMaopera.estado,1)='0'
	and LEFT(CsrMaopera.clasecomp,1)<>'K'
	and CsrMovCtacte.fecha > '19860201' and LEFT(CsrMovCtacte.switch,3) like '__0'
	and not (CsrComprobante.id in (8))
	and CsrMovCtacte.id >0
	order by csrmovctacte.vencimien,csrmovctacte.id
