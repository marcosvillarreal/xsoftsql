			SELECT 0 as ok,ISNULL(CsrCtacte.cnumero,'') as Csrcodigo,ISNULL(Csrcabedeta.detalle,ISNULL(CsrCtacte.cnombre,space(30))) as Csrnombre,Csrmaopera.fechasis as fecha,'          ' as Csrfecha
			,Csrcabeord.total as Csrtotal,CsrCabeord.signo,Csrmaopera.id as idmaopera,Csrmaopera.idcomproba as idcomproba
			,ISNULL(Csrcomprobante.cnombre,SPACE(30)) as Csrnomcomp,Csrmaopera.estado,Csrmaopera.clasecomp
			,LEFT(Csrmaopera.numcomp,1)+' '+SUBSTRING(Csrmaopera.numcomp,2,4)+'-'+RIGHT(Csrmaopera.numcomp,8) as Csrnrocomp,Csrmaopera.origen
		    FROM maopera as Csrmaopera
		    left join cabeord as Csrcabeord on Csrmaopera.id = Csrcabeord.idmaopera
		    left join cabedeta as Csrcabedeta on Csrmaopera.id = Csrcabedeta.idmaopera
		    left join comprobante as Csrcomprobante on Csrmaopera.idcomproba = Csrcomprobante.id
			left join ctacte as CsrCtacte on CsrCabeord.idctacte = CsrCtacte.id
			WHERE Csrmaopera.clasecomp<>'K' and (CSRMAOPERA.FECHASIS='20160623' AND isnull(CSRCABEDETA.DETALLE,SPACE(40)) LIKE '%%') and CsrMaopera.idcomproba IN (9,10,1100000051,1100000055) and Csrmaopera.estado<>'1' and Csrmaopera.origen='STK'
			and LEFT(ISNULL(CsrCabeDeta.switch,'0'),1)='0'
			order by CsrMaopera.fechasis
