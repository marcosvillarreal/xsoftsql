SELECT Csrproducto.*,ISNULL(Tipoiva.tasa,0) as tasa,ISNULL(rubro.idfuerzavta,0) as idfuerzavta,isnull(left(tipoprod.switch,1),'0') as alcohol,ISNULL(rubro.perceibruto,0) as perceibruto,isnull(rubro.omitirlistas,0) as omitirlistas FROM Producto as Csrproducto LEFT JOIN Tipoiva on Csrproducto.idiva = Tipoiva.id LEFT JOIN rubro on Csrproducto.idrubro = rubro.id LEFT JOIN tipoprod on rubro.idtipoprod = tipoprod.id where Csrproducto.numero=100 AND nofactura=0 and idestado=1
