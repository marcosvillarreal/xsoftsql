SELECT CsrCtacte.id as idctacte,CsrCtacte.cnumero as codcliente,CsrLocalidad.cpostal
,CsrLocalidad.nombre as nomlocalidad,CsrCateCtacte.nombre as nomcanal
,CsrCtacte.estadocta as status,CsrCtacte.cnombre as nomcliente
,CsrCtacte.cuit,CsrCtacte.dni,CsrCtacte.cdireccion as direccion
,CsrCategoiva.abrevia,'' as nomzona,'' as diasvisita,'' as frecvisita
,(Select top 1 CsrVendedor.nombre  From CabeFac as CsrCabefac
	inner join Maopera as CsrMaopera on CsrCabefac.idmaopera = CsrMaopera.id
	inner join Vendedor as CsrVendedor on CsrMaopera.idvendedor = CsrVendedor.id
	Order By CsrMaopera.id desc) as nomvendedor
,'Propia' as TipoCigarrera,1 as cantcigarrera
,CsrCtacte.ctelefono as telefono
From Ctacte as CsrCtacte
left join Categoiva as CsrCategoiva on CsrCtacte.tipoiva = CsrCategoiva.id
left join Localidad as CsrLocalidad on CsrCtacte.idlocalidad = CsrLocalidad.id
left join CateCtacte as CsrCateCtacte on CsrCtacte.idcategoria = CsrCateCtacte.id
Where CsrCtacte.ctadeudor = 1
