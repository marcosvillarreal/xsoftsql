			SELECT 0 as ok,Csrcabefac.ctacte as Csrcodigo,Csrcabefac.cnombre as Csrnombre,Csrcabefac.fecha as fecha
			,CONVERT(char(10),CsrCabefac.fecha,105) as Csrfecha
			,Csrcabefac.total as Csrtotal,CsrCabefac.signo,Csrmaopera.id as idmaopera,Csrmaopera.idcomproba as idcomproba
			,ISNULL(Csrcomprobante.cnombre,SPACE(30)) as Csrnomcomp,Csrmaopera.estado,Csrmaopera.clasecomp
			,LEFT(Csrmaopera.numcomp,1)+' '+SUBSTRING(Csrmaopera.numcomp,2,4)+'-'+RIGHT(Csrmaopera.numcomp,8) as Csrnrocomp,Csrmaopera.origen
		    FROM maopera as Csrmaopera
		    left join cabefac as Csrcabefac on Csrmaopera.id = Csrcabefac.idmaopera
		    left join comprobante as Csrcomprobante on Csrmaopera.idcomproba = Csrcomprobante.id
			WHERE not Csrmaopera.clasecomp in ('K','0') and (CSRCABEFAC.CTACTE='100' AND CSRMAOPERA.NUMCOMP LIKE '%%') and Csrmaopera.estado<>'1' and CsrMaopera.idcomproba IN (1,3,2) and (Csrmaopera.origen='FAC' or Csrmaopera.origen='FPE') order by csrmaopera.numcomp
