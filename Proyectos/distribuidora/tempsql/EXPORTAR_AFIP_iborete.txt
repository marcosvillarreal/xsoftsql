	SELECT Csrmaopera.numcomp as numcomp,Csrmaopera.estado
	,Convert(varchar(8),Csrmovctacte.fecha,112) as chrfecha
	,Convert(varchar(10),Csrmovctacte.fecha,103) as fecha_fac
	,CsrCtacte.cnombre as cnombre,CsrCtacte.cuit as cuit
	,ISNULL(Csrtablaimp.tasa,00.00) as tasa,CsrCtacte.inscri02 as Nroibto
	,CsrProvincia.numero as CodProv, ISNULL(Csrlocalidad.nombre,SPACE(30)) as Localidad
	,CsrCtacte.cdireccion as cdireccion,CsrLocalidad.cpostal as cpostal
		,ISNULL(CsrTablaimp.detalle,SPACE(20)) as numrete, CsrMaopera.clasecomp
	,(Case when Csrtablaimp.tipoconce='PB' and  CsrParaVario.nombre = 'CATEIBBA' 
		then ISNULL(Csrtablaimp.baseimp * CsrMovctacte.signo*1,cast(0 as numeric(12,2))) 
		else cast(0 as numeric(12,2)) end) as baseimpPB
	,(Case when Csrtablaimp.tipoconce='RB' and  CsrParaVario.nombre = 'CATEIBBA' 
		then ISNULL(Csrtablaimp.baseimp * CsrMovctacte.signo*1,cast(0 as numeric(12,2))) 
		else cast(0 as numeric(12,2)) end) as baseimpRB
	,(Case when Csrtablaimp.tipoconce='PB' and  CsrParaVario.nombre = 'CATEIBBA' 
		then ISNULL(Csrtablaimp.importe * CsrMovctacte.signo*1,cast(0 as numeric(12,2))) 
		else cast(0 as numeric(12,2)) end) as TotalPB
	,(Case when Csrtablaimp.tipoconce='RB' and  CsrParaVario.nombre = 'CATEIBBA' 
		then ISNULL(Csrtablaimp.importe * CsrMovctacte.signo*1,cast(0 as numeric(12,2))) 
		else cast(0 as numeric(12,2)) end) as TotalRB
	,(Case when Csrtablaimp.tipoconce='RB' and  CsrParaVario.nombre = 'CATEIBBA' 
		then ISNULL(Csrtablaimp.tasa,cast(0 as numeric(12,2))) 
		else cast(0 as numeric(12,2)) end) as TasaRB
	,(Case when Csrtablaimp.tipoconce='PB' and  CsrParaVario.nombre = 'CATEIBBA' 
		then ISNULL(Csrtablaimp.tasa,cast(0 as numeric(12,2))) 
		else cast(0 as numeric(12,2)) end) as TasaPB
	,isnull(CsrCategoibto.porrete,-1) as  tasactacte
	,CsrCtacte.convenio
	,ISNULL(CsrProvIBTO.alicuota4,1) as tasaprov
	from maopera as Csrmaopera
	left join movctacte as CsrmovCtacte on CsrMaopera.id = CsrMovctacte.idmaopera
	left join ctacte as Csrctacte on CsrMovctacte.idctacte = Csrctacte.id
	left join categoiva as Csrcategoiva on CsrCtacte.tipoiva = Csrcategoiva.id
	left join catectacte as Csrcatectacte on CsrCtacte.idcategoria = CsrCateCtacte.id
	left join comprobante as Csrcomprobante on Csrmaopera.idcomproba = Csrcomprobante.id
	left join provincia as csrprovincia on CsrCtacte.idprovincia = Csrprovincia.id
	left join localidad as CsrLocalidad on CsrCtacte.idlocalidad = CsrLocalidad.id
	left join tablaimp as csrtablaimp on csrmaopera.id = csrtablaimp.idmaopera 
	left join paravario as CsrParavario on CsrTablaimp.idprovincia = CsrParavario.idorigen
	LEFT JOIN CATEIBBA AS CsrCategoibto ON  replace(replace(csrctacte.cuit,'-',''),'/','') = CsrCategoibto.cuit
	left join Provincia as CsrProvIBTO on CsrTablaimp.idprovincia = CsrProvIBTO.id
	where (CsrMovctacte.fecha BETWEEN '20160831' and '20160831') and Csrmaopera.origen IN ('COB') and Csrmaopera.clasecomp <>'K' 
	and Csrtablaimp.tipoconce in ('RB') and CsrMaopera.estado='0' and right(CsrMaopera.numcomp,8)<>''
	and ISNULL(CsrParavario.idorigen,0)>0
