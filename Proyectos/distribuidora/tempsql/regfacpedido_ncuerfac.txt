SELECT CsrNcuerfac.*
,ISNULL(Csrproducto.idctavta,0) as idctavta,ISNULL(Csrproducto.idenvase,0) as idenvase
,ISNULL(CsrProducto.numero,CAST(0 as int)) as CodProd
,ISNULL(CsrRubro.numero,CAST(0 as int)) as codrubro
,ISNULL(CsrRubro.id,CAST(0 as int)) as idrubro
from NCuerfac as CsrNcuerfac
left join producto as Csrproducto on CsrNcuerfac.idarticulo = Csrproducto.id
left join rubro as CsrRubro on CsrProducto.idrubro = CsrRubro.id
where CsrNcuerfac.idmaopera in (110000005126,110000005127)
order by CsrNCuerfac.idmaopera,CsrProducto.numero,CsrRubro.numero
