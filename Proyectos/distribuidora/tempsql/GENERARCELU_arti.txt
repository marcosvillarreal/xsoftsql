SELECT  csrProducto.*, ISNULL(CsrRubro.numero,999) as seccion
,ISNULL(CsrTipoiva.tasa,99) as iva, ISNULL(CsrMarca.numero,0000) as marca
,ISNULL(Csrctacte.cnumero,000000) as proveedor
,ISNULL(CsrForma.numero,00000) as envase
,ISNULL(CsrTipoiva.numero,0000) as tipoiva
,ISNULL((select top 1 CsrSubproducto.idvariedad from Subproducto as CsrSubproducto
Where CsrSubproducto.idarticulo = CsrProducto.id and CsrSubProducto.nofactura=0 and CsrSubProducto.estado=0),0) as variedad
,ISNULL(CsrFuerzaVta.numero,1) as numfuerzavta
FROM Producto as CsrProducto
left join rubro as CsrRubro on CsrProducto.idrubro = CsrRubro.id
left join Tipoiva as CsrTipoiva on CsrProducto.idiva = CsrTipoiva.id
left join marca as Csrmarca on CsrProducto.idmarca = CsrMarca.id
left join ctacte as Csrctacte on CsrProducto.idctacte = Csrctacte.id
left join forma as Csrforma on csrProducto.idenvase = Csrforma.id
left join fuerzavta as CsrFuerzaVta on CsrRubro.idfuerzavta = CsrFuerzaVta.id
where CsrProducto.idestado=1 and csrproducto.nofactura=0 and csrproducto.nolista in (0)
 and CsrProducto.prevta1<>0 
 and CsrProducto.espromocion = 0
 order by CsrRubro.numero,CsrProducto.numero
